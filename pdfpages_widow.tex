%% start of file `pdfpages_widow.tex'.
% DG
% A file, say foo.pdf, has 2 pages, the second page only contains a 
% short piece of text which can fit on page 1 by reducing the upper border.
%trim=left botm right top
\documentclass[11pt,a4paper]{article}
\usepackage{ngerman,eso-pic,graphicx}%helvet or times
\usepackage{pdfpages}
\begin{document}
\thispagestyle{empty}
\AddToShipoutPicture{%
	\setlength{\unitlength}{1mm}%
	\put(0,7){%
	% This includes the content of page 2 at the bottom of page 1
	\includegraphics[page=2,width=210mm,clip, trim=0 285mm 0 6mm]{foo.pdf}
	% you may measure location of content on page2 with ghostview
	}}
\includepdf[pages=1,trim=0 0 0 9mm]{foo.pdf}% trim page at the top
%
\end{document} pdfpages_widow.tex
