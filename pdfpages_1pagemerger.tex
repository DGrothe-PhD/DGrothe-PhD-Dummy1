%LaTeX2e
%% A simple macro for combining three 1-page pdf files,
%% for example saved by a scanner,
%% into one pdf file
%% wherein the pdfs have the same name except from a suffix.
%
\newcommand{\zamm}[4][]{%
\includepdf[pages=1]{#1#2.pdf}
\includepdf[pages=1]{#1#3.pdf}
\includepdf[pages=1]{#1#4.pdf}}

%Example usage
\documentclass[11pt,a4paper]{article}
\usepackage{pdfpages}
\begin{document}
%Example for IMG_20200213_0001 to IMG_20200213_0003
\zamm[IMG_20200213_000]{1}{2}{3}
%
\end{document}
